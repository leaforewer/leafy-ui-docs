classnames:
  component:
    - class: leaf-carousel
      desc: Main carousel web component container
  attributes:
    - class: snap
      desc: Enable snap-to-slide behavior
    - class: infinite
      desc: Enable infinite scrolling
    - class: drag-sensitivity
      desc: Controls drag responsiveness (number, default 0.004)
    - class: lerp-factor
      desc: Smoothness of transitions (number, default 0.28)
    - class: snap-strength
      desc: Strength of snap effect (number, default 0.08)
    - class: speed-decay
      desc: Scroll momentum decay rate (number, default 0.88)

import { Tabs } from 'nextra/components'

# Carousel

Interactive carousel component with smooth dragging, infinite scrolling, and customizable animations. Built as a web component for framework-agnostic usage.

## Setup

Choose your setup method based on your framework:

<Tabs items={['JavaScript']}>
  <Tabs.Tab>
    First, install the Leafy UI core package:
    
    ```bash
    npm install @leafy-ui/core
    ```
    
    Then import the carousel web component in your project:
    
    ```js
    import '@leafy-ui/core/components/leaf-carousel'
    ```
  </Tabs.Tab>
</Tabs>

### Basic carousel

Carousel comes default with snap and infinite scroll, so you don't need to add anything just add ```<leaf-carousel>```.
